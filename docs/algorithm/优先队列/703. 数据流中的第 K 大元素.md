class Heap {
    constructor(heap = [], compareFn = (a, b) => a - b) {
        this.heap = heap;
        // 比较器，区分最大堆和最小堆，默认 最小堆
        this.compareFn = compareFn;
        this.heapify(); // 堆化
    }
    getLeftIndex(index) {
        return 2 * index + 1;
    }

    getRightIndex(index) {
        return 2 * index + 2;
    }

    getParentIndex(index) {
        if (index === 0) {
            return undefined;
        }
        return Math.floor((index - 1) / 2);
    }

    size() {
        return this.heap.length;
    }
    isEmpty() {
        return this.size() <= 0;
    }
    clear() {
        this.heap = [];
    }

    heapify() {
        if (this.size() <= 1) {
            return;
        }
        let lastIndex = this.size() - 1;
        let parentIndex = this.getParentIndex(lastIndex);
        // 从最后一个节点的父节点开始做 heapify
        for (let i = parentIndex; i >= 0; i--) {
            this.siftDown(i);
        }
    }
    // 这个方法返回最小值(最小堆)或最大值(最大堆)且不会移除这个值
    peek() {
        if (this.size() === 0) {
            return null;
        }
        return this.heap[0];
    }
    /**
     * 这个方法移除最小值(最小堆)或最大值(最大堆)，并返回这个值
     * 1. 交换根节点与最后一个节点的值
     * 2. 删除最后一个节点
     * 3. 把根节点向下调整
     * 时间复杂度O(logn)
     */
    poll() {
        if (this.size() === 0) {
            return null;
        }
        if (this.size() === 1) {
            return this.heap.shift();
        }
        const removedValue = this.heap[0];
        const last = this.heap.pop();
        if (this.size !== 0) {
            this.heap[0] = last; // 交换第一个元素和最后一个元素
            this.siftDown(0);
        }
        return removedValue;
    }
    /**
     * 向堆中添加元素
     * 1. 把新数据添加到树的最后一个元素，也就是数组的末尾
     * 2. 把末尾节点向上调整，siftUp
     * 时间复杂度O(logn)
     */
    insert(value) {
        if (value !== null) {
            const index = this.heap.length;
            // 先加到最后，然后再上移
            this.heap.push(value);
            this.siftUp(index);
            return true;
        }
        return false;
    }

    // 上移操作，从当前节点的父元素开始进行比较
    siftUp(index) {
        while (index > 0) {
            let parentIndex = this.getParentIndex(index);
            // 如果当前元素比父节点的元素小，就交换当前节点和父节点的位置
            if (this.compareFn(this.heap[index], this.heap[parentIndex]) < 0) {
                this.swap(index, parentIndex);
                // 不断向上取父节点进行比较
                index = parentIndex;
            } else {
                break;
            }
        }
    }
    // 下移操作(堆化)
    siftDown(index) {
        let lastIndex = this.size() - 1;
        const left = this.getLeftIndex(index);
        const right = this.getRightIndex(index);
        let findIndex = index;
        if (left <= lastIndex && this.compareFn(this.heap[left], this.heap[findIndex]) < 0) {
            findIndex = left;
        }
        if (right <= lastIndex && this.compareFn(this.heap[right], this.heap[findIndex]) < 0) {
            findIndex = right;
        }
        if (index !== findIndex) {
            this.swap(index, findIndex);
            this.siftDown(findIndex);
        }
    }

    swap(i, j) {
        [this.heap[i], this.heap[j]] = [this.heap[j], this.heap[i]];
    }
}
/**
 * @param {number} k
 * @param {number[]} nums
 */
var KthLargest = function(k, nums) {
    this.k = k;
    this.heap = new Heap();
    for (let x of nums) {
        this.add(x);
    }
};

/**
 * @param {number} val
 * @return {number}
 */
KthLargest.prototype.add = function(val) {
    this.heap.insert(val);
    if (this.heap.size() > this.k) {
        this.heap.poll();
    }
    return this.heap.peek();
};
